🚀 Quick Setup

# For Ubuntu 22.04+ / Debian 11+
curl -fsSL https://Silica96.github.io/deb-test/public.key | \
  sudo gpg --dearmor -o /etc/apt/keyrings/deb-test.gpg

echo "deb [signed-by=/etc/apt/keyrings/deb-test.gpg] \
  https://Silica96.github.io/deb-test/ stable main" | \
  sudo tee /etc/apt/sources.list.d/deb-test.list

sudo apt update
sudo apt install test
